% Overwrite / adjust existing LaTeX commands.
% set vectors in bold face, without arrow accent
\renewcommand{\vec}[1]
{
    \mathbf{#1}
}
% use bold face plus hat / roof accent for unit vectors
\let\oldhat\hat
\renewcommand{\hat}[1]
{
    \oldhat{\mathbf{#1}}
}
% set differentials in upright font
\newcommand*\diff
{
    \mathrm{d}
}

\stackMath                              %  change the default to process arguments in inline math mode
\newcommand\vwidehat[1]%
{%
    \savestack
    {\tmpbox}
    {%
        \stretchto%         % create and save an intermediate result
        {
            \scaleto%
            {%
                \scalerel*[\widthof{\ensuremath{#1}}]%
                {\kern-.6pt\bm\wedge\kern-.6pt}%
                {\rule[-\textheight/2]{1ex}{\textheight}}%
            }
            {\textheight}%
        }%
        {0.55ex}%
    }%
    \stackon[1pt]{#1}{\tmpbox}%
}
\newcommand\usim[2][1]%
{%
    \ifnum#1>1%
        \stackunder[0pt]{\usim[\numexpr#1-1\relax]{#2}}{\scriptscriptstyle\sim}%
    \else%
        \stackunder[1pt]{#2}{\scriptscriptstyle\sim}%
    \fi%
}
